# ingest.py
# pip install pymupdf tiktoken supabase openai tqdm python-dotenv

import os, uuid, re
import fitz  # PyMuPDF
import tiktoken
from supabase import create_client, Client
from openai import OpenAI
from tqdm import tqdm
from dotenv import load_dotenv, find_dotenv

# ── Load environment ──────────────────────────────────────────────────────────
load_dotenv(find_dotenv(usecwd=True))

SUPABASE_URL             = os.environ["SUPABASE_URL"]
SUPABASE_SERVICE_ROLE_KEY = os.environ["SUPABASE_SERVICE_ROLE_KEY"]
OPENAI_API_KEY           = os.environ["OPENAI_API_KEY"]

# ── Config ────────────────────────────────────────────────────────────────────
PDF_PATH        = "human-nutrition-text.pdf"
DOC_ID          = "nutrition-v1"          # keep STABLE to avoid duplicates
EMBED_MODEL     = "text-embedding-3-small" # 1536 dims → matches your table
BATCH_EMBED     = 100
BATCH_INSERT    = 200

# Sentence chunking params
SENTS_PER_CHUNK = 10
SENT_OVERLAP    = 2
MAX_TOKENS      = 1300   # safety cap (trim if 10 sentences are too long)
MIN_TOKENS      = 50     # skip very tiny fragments

enc = tiktoken.get_encoding("cl100k_base")

# ── Clients ───────────────────────────────────────────────────────────────────
supabase: Client = create_client(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY)
oai = OpenAI(api_key=OPENAI_API_KEY)

# ── Helpers ───────────────────────────────────────────────────────────────────
def extract_text_from_pdf(path: str) -> list[dict]:
    """Return list of {page_number, text} dicts."""
    doc = fitz.open(path)
    pages = []
    for i, page in enumerate(doc):
        text = page.get_text().replace("\n", " ").strip()
        if text:
            pages.append({"page_number": i, "text": text})
    doc.close()
    return pages

def split_into_sentences(text: str) -> list[str]:
    """Naive but fast sentence splitter."""
    raw = re.split(r'(?<=[.!?])\s+', text)
    return [s.strip() for s in raw if s.strip()]

def make_chunks(sentences: list[str]) -> list[str]:
    """Sliding-window sentence chunks with overlap."""
    chunks = []
    step = SENTS_PER_CHUNK - SENT_OVERLAP
    for start in range(0, len(sentences), step):
        window = sentences[start: start + SENTS_PER_CHUNK]
        chunk  = " ".join(window)
        tokens = len(enc.encode(chunk))
        if tokens < MIN_TOKENS:
            continue
        if tokens > MAX_TOKENS:
            # trim greedily
            trimmed, acc = [], 0
            for s in window:
                t = len(enc.encode(s))
                if acc + t > MAX_TOKENS:
                    break
                trimmed.append(s)
                acc += t
            chunk = " ".join(trimmed)
        chunks.append(chunk)
        if start + SENTS_PER_CHUNK >= len(sentences):
            break
    return chunks

def embed_batch(texts: list[str]) -> list[list[float]]:
    resp = oai.embeddings.create(model=EMBED_MODEL, input=texts)
    return [r.embedding for r in resp.data]

def upsert_chunks(rows: list[dict]):
    supabase.table("chunks").upsert(rows, on_conflict="doc_id,chunk_index").execute()

# ── Main pipeline ─────────────────────────────────────────────────────────────
def main():
    print(f"📄  Extracting text from {PDF_PATH} …")
    pages = extract_text_from_pdf(PDF_PATH)

    print("✂️   Building sentence chunks …")
    all_chunks: list[dict] = []
    global_idx = 0
    for page in pages:
        sentences = split_into_sentences(page["text"])
        for chunk_text in make_chunks(sentences):
            all_chunks.append({
                "doc_id":      DOC_ID,
                "chunk_index": global_idx,
                "content":     chunk_text,
                "metadata": {
                    "page": page["page_number"],
                    "source": PDF_PATH,
                },
            })
            global_idx += 1

    print(f"   → {len(all_chunks)} chunks ready")

    print("🔢  Embedding …")
    texts = [c["content"] for c in all_chunks]
    embeddings: list[list[float]] = []
    for i in tqdm(range(0, len(texts), BATCH_EMBED)):
        embeddings.extend(embed_batch(texts[i: i + BATCH_EMBED]))

    for chunk, emb in zip(all_chunks, embeddings):
        chunk["embedding"] = emb

    print("💾  Upserting to Supabase …")
    for i in tqdm(range(0, len(all_chunks), BATCH_INSERT)):
        upsert_chunks(all_chunks[i: i + BATCH_INSERT])

    print(f"✅  Done — {len(all_chunks)} chunks stored under doc_id='{DOC_ID}'")

if __name__ == "__main__":
    main()

// supabase/functions/voice-log/index.ts
import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const SARVAM_API_KEY    = Deno.env.get("SARVAM_API_KEY")!
const ANTHROPIC_API_KEY = Deno.env.get("ANTHROPIC_API_KEY")!
const OPENAI_API_KEY    = Deno.env.get("OPENAI_API_KEY")!

serve(async (req) => {
  const { audio_base64, language_code, user_id } = await req.json()

  // Step 1: Sarvam STT + Translation
  const sarvamResp = await fetch("https://api.sarvam.ai/speech-to-text-translate", {
    method: "POST",
    headers: {
      "api-subscription-key": SARVAM_API_KEY,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      audio: audio_base64,
      language_code: language_code || "unknown",
      target_language_code: "en-IN"
    })
  })
  const { transcript, translated_text } = await sarvamResp.json()

  // Step 2: Claude extracts food items
  const claudeExtract = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "x-api-key": ANTHROPIC_API_KEY,
      "anthropic-version": "2023-06-01",
      "content-type": "application/json"
    },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 500,
      messages: [{
        role: "user",
        content: `Extract food items and quantities from this text. Return JSON only.
Text: "${translated_text}"
Format: {"items": [{"name": "food name", "quantity": 1, "unit": "piece/cup/gram/etc", "meal_type": "breakfast/lunch/dinner/snack"}]}`
      }]
    })
  })
  const extractData = await claudeExtract.json()
  const foodItems = JSON.parse(extractData.content[0].text)

  // Step 3: RAG query for each food item
  const embResp = await fetch("https://api.openai.com/v1/embeddings", {
    method: "POST",
    headers: { "Authorization": `Bearer ${OPENAI_API_KEY}`, "Content-Type": "application/json" },
    body: JSON.stringify({ model: "text-embedding-3-small", input: translated_text })
  })
  const { data: [{ embedding }] } = await embResp.json()

  // Query Supabase
  const { createClient } = await import("https://esm.sh/@supabase/supabase-js@2")
  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  )

  const { data: chunks } = await supabase.rpc("match_documents", {
    query_embedding: embedding,
    match_count: 8
  })

  const context = chunks.map((c: any) => c.content).join("\n\n")

  // Step 4: Claude generates nutrition data using RAG context
  const claudeNutrition = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "x-api-key": ANTHROPIC_API_KEY,
      "anthropic-version": "2023-06-01",
      "content-type": "application/json"
    },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 1000,
      messages: [{
        role: "user",
        content: `Using the nutrition reference below, calculate nutritional values for these foods.
        
Foods: ${JSON.stringify(foodItems.items)}

Reference: ${context}

Return JSON: {
  "total": {"calories": 0, "protein_g": 0, "carbs_g": 0, "fat_g": 0, "fiber_g": 0},
  "per_item": [{"name": "", "calories": 0, "protein_g": 0, "carbs_g": 0, "fat_g": 0}],
  "summary": "brief 1-sentence friendly summary"
}`
      }]
    })
  })

  const nutritionData = await claudeNutrition.json()
  const nutrition = JSON.parse(nutritionData.content[0].text)

  // Step 5: Store in Supabase (unconfirmed — user must confirm)
  const { data: mealLog } = await supabase
    .from("meal_logs")
    .insert({
      user_id,
      raw_voice_input: transcript,
      translated_input: translated_text,
      food_items: foodItems.items,
      nutrition_data: nutrition,
      confirmed: false,
      meal_type: foodItems.items[0]?.meal_type || "snack"
    })
    .select()
    .single()

  return new Response(JSON.stringify({
    meal_log_id: mealLog.id,
    transcript,
    translated_text,
    food_items: foodItems.items,
    nutrition,
    needs_confirmation: true
  }), { headers: { "Content-Type": "application/json" } })
})

// supabase/functions/confirm-meal/index.ts
import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from "https://esm.sh/@supabase/supabase-js@2"

serve(async (req) => {
  const { meal_log_id, user_id, edited_items } = await req.json()
  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  )

  // Mark as confirmed, optionally update items
  const updateData: any = { confirmed: true }
  if (edited_items) updateData.food_items = edited_items

  const { data: meal } = await supabase
    .from("meal_logs")
    .update(updateData)
    .eq("id", meal_log_id)
    .select()
    .single()

  // Upsert daily summary
  const n = meal.nutrition_data.total
  await supabase.rpc("increment_daily_summary", {
    p_user_id: user_id,
    p_date: new Date().toISOString().split("T")[0],
    p_calories: n.calories,
    p_protein:  n.protein_g,
    p_carbs:    n.carbs_g,
    p_fat:      n.fat_g,
    p_fiber:    n.fiber_g
  })

  return new Response(JSON.stringify({ success: true, meal }), {
    headers: { "Content-Type": "application/json" }
  })
})

    query_embedding: embedding, match_count: 6
  })

  // Get user's daily summary for personalization
  const { data: dailySummary } = await supabase
    .from("daily_summaries")
    .select("*")
    .eq("user_id", user_id)
    .eq("date", new Date().toISOString().split("T")[0])
    .single()

  const context = chunks.map((c: any) => c.content).join("\n\n")

  const systemPrompt = `You are NutriVoice, a friendly nutrition assistant.
Use the reference material below to answer questions accurately.
Today the user has consumed: ${JSON.stringify(dailySummary || {})}.
Be concise, warm, and actionable. Reference specific numbers when relevant.

NUTRITION REFERENCE:
${context}`

  const claudeResp = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "x-api-key": Deno.env.get("ANTHROPIC_API_KEY"),
      "anthropic-version": "2023-06-01",
      "content-type": "application/json"
    },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 800,
      system: systemPrompt,
      messages: [
        ...conversation_history,
        { role: "user", content: message }
      ]
    })
  })

  const result = await claudeResp.json()
  const reply = result.content[0].text

  // Save to chat history
  await supabase.from("chat_messages").insert([
    { user_id, role: "user", content: message },
    { user_id, role: "assistant", content: reply }
  ])

  return new Response(JSON.stringify({ reply }), {
    headers: { "Content-Type": "application/json" }
  })
